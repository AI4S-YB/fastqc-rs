  sub poly_wildcard{                                                            
      # my $wild = shift;                                                       
      my $s = shift;                                                            
      my $wild = 'A';                                                           
      my $n = length($s);                                                       
      my $best_index = $n;                                                      
      my $best_score = my $score = my $errors = 0;                              
      for my $i (reverse(0..$n-1)) {                                            
          my $nuc = substr($s, $i, 1);                                          
          if ($nuc eq $wild) {                                                  
              $score++;                                                         
          } else {                                                              
              $score -= 2;                                                      
              $errors++;                                                        
          }                                                                     
          if ($score > $best_score && $errors <= 0.2 * ($n - $i)) {             
              $best_index = $i;                                                 
              $best_score = $score;                                             
          }                                                                     
      }                                                                         
      $s = substr($s, 0, $best_index);                                          
                                                                                
  }

  ---
  Summary

  Added a new trim_galore module as a subcommand (fastqc-rs trim-galore) that
  wraps Cutadapt for adapter/quality trimming, matching the functionality of the
   Perl Trim Galore tool.

  Files created:

  - src/trim_galore/config.rs - CLI configuration with all Trim Galore options
  (clap Parser)
  - src/trim_galore/adapter.rs - Adapter auto-detection (scans first 1M reads
  for Illumina/Nextera/smallRNA)
  - src/trim_galore/mod.rs - Main trimming logic

  Files modified:

  - src/lib.rs - Added pub mod trim_galore
  - src/main.rs - Added trim-galore subcommand dispatch (backward-compatible,
  existing QC mode unchanged)

  Key features:

  - Cutadapt path specification: --path-to-cutadapt /path/to/cutadapt
  - Single-end & paired-end trimming with validation
  - Adapter auto-detection (Illumina, Nextera, smallRNA) + presets (--illumina,
  --nextera, --small-rna, --stranded-illumina, --bgiseq)
  - Quality trimming via cutadapt (-q, --nextseq)
  - Hard-trim modes (--hardtrim5, --hardtrim3)
  - Clipping (--clip_R1/R2, --three_prime_clip_R1/R2)
  - Filtering: min/max length, max N content
  - RRBS support (auto --clip_R2 2 for paired-end directional)
  - Output handling: gzip, output dir, basename rename, reports
  - FastQC integration (--fastqc)
  - Multi-core support via cutadapt's -j flag

  Usage:

  # Single-end with auto-detection
  fastqc-rs trim-galore reads.fq.gz

  # Paired-end with custom cutadapt path
  fastqc-rs trim-galore --paired --path-to-cutadapt /opt/cutadapt \
    R1.fq.gz R2.fq.gz

  # With specific adapter and options
  fastqc-rs trim-galore --illumina -q 30 --length 50 -j 4 \
    -o trimmed/ reads.fq.gz
